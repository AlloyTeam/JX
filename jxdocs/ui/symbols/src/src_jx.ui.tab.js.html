1 /**
  2  * tab模块
  3  */
  4 Jx().$package(function(J){
  5     var $ = J.dom.id,
  6         $D = J.dom,
  7         $E = J.event;
  8     J.ui = J.ui || {};
  9     /**
 10      * Tab类
 11      * 
 12      * @memberOf ui
 13      * @name Tab
 14      * @class
 15      * @constructor
 16      * @param {Element} triggers tab head元素
 17      * @param {Element} sheets tab body元素
 18      * @param {Object} config 其他选项，如:isLimited,leftMargin...
 19      * 
 20      */
 21     J.ui.Tab = function(triggers,sheets,config){
 22         this.tabs = [];             //tab的集合
 23         this.currentTab = null;     //当前tab
 24         this.config = {
 25             defaultIndex : 0,       //默认的tab索引
 26             triggerEvent : 'click', //默认的触发事件
 27             slideEnabled : false,   //是否自动切换
 28             slideInterval : 5*1000,   //切换时间间隔
 29             slideDelay : 300,       //鼠标离开tab继续切换的延时
 30             autoInit : true,        //是否自动初始化
 31             /** @ignore */
 32             onShow:function(){ }    //默认的onShow事件处理函数
 33         };
 34     
 35         this.setConfig(config);
 36 
 37         if(triggers && sheets) {
 38             this.addRange(triggers,sheets);
 39             if(this.config.autoInit){
 40                 this.init();
 41             }
 42         }
 43     };
 44     
 45     J.ui.Tab.prototype = 
 46     /**
 47      * @lends ui.Tab.prototype 
 48      */
 49     {
 50         /**
 51          * 初始化函数
 52          * @ignore
 53          *
 54          **/
 55         init:function(){
 56             var config = this.config,
 57                 _this = this;
 58 
 59             J.array.forEach(this.tabs, function(tab, index, tabs){
 60                 $E.on(tab.trigger,config['triggerEvent'], function(){
 61                     _this.select.call(_this,tab);
 62                 });
 63                 if(tab.sheet){
 64                     tab.sheet.style.display = 'none';
 65                 }
 66             });
 67             
 68             this.select(this.tabs[config['defaultIndex']]);
 69             if(config['slideEnabled']){
 70                 this.slide();
 71             }
 72         },
 73         /**
 74          * 设置config
 75          * @param {Object} config 配置项如{'slideEnabled':true,'defaultIndex':0,'autoInit':false}
 76          */
 77         setConfig:function(config){
 78             if(!config) return;
 79             for(var i in config){
 80                 this.config[i] = config[i];
 81             }
 82         },
 83         /**
 84          * 增加tab
 85          * @param {Object} tab {trigger:aaaa, sheet:bbbb} 
 86          * trigger: {String},{HTMLElement}
 87          * sheet: {String},{HTMLElement}
 88          * */
 89         add:function(tab){
 90             if(!tab) return;
 91             
 92             if(tab.trigger){
 93                 this.tabs.push(tab);
 94                 tab.trigger.style.display = 'block';
 95             }
 96         },
 97         
 98         /**
 99          * 增加tab数组
100          * @param {Array} triggers HTMLElement 数组
101          * @param {Array} sheets HTMLElement 数组
102          * */
103         addRange:function(triggers, sheets){
104             if(!triggers||!sheets) return;
105             if(triggers.length && sheets.length && triggers.length == sheets.length){
106                 for(var i = 0, len = triggers.length; i < len; i++){
107                     this.add({trigger:triggers[i],sheet:sheets[i]});
108                 }
109             }
110         },
111         
112         /**
113          * 设置tab为当前tab并显示
114          * @param {Object} tab  tab对象 {trigger:HTMLElement,sheet:HTMLElement}
115          * */
116         select:function(tab){
117             if(!tab || (!!this.currentTab && tab.trigger == this.currentTab.trigger)) return;
118             if(this.currentTab){
119                 $D.removeClass(this.currentTab.trigger, 'current');
120                 if(this.currentTab.sheet){
121                     this.currentTab.sheet.style.display = 'none';
122                 }
123                 
124             }
125             this.currentTab = tab;
126             this.show();
127         },
128         
129         /**
130          * 设置tab为隐藏的
131          * @param {Object} tab  tab对象 {trigger:HTMLElement,sheet:HTMLElement}
132          * */
133         remove:function(tab){
134             if(!tab) return;
135             
136             
137             if(tab.trigger){
138                 $D.removeClass(tab.trigger, 'current');
139                 tab.trigger.style.display = 'none';
140             }
141             if(tab.sheet){
142                 tab.sheet.style.display = 'none';
143             }
144             
145             var index=this.indexOf(tab);
146             this.tabs.splice(index,index);
147     
148             if(tab.trigger == this.currentTab.trigger){
149                 if(index==0){
150                     //this.currentTab=this.tabs[(index+1)];
151                     this.select(this.tabs[(index+1)]);
152                 }else{
153                     //this.currentTab=this.tabs[(index-1)];
154                     this.select(this.tabs[(index-1)]);
155                 }
156             }
157         },
158         /**
159          * 显示当前被选中的tab
160          * */
161         show:function(){
162             
163             if(this.currentTab.trigger){
164                 this.currentTab.trigger.style.display = 'block';
165             }
166             $D.addClass(this.currentTab.trigger, 'current');
167             if(this.currentTab.sheet){
168                 this.currentTab.sheet.style.display = 'block';
169             }
170             //触发自定义显示对象
171             //tealin
172             this.config.onShow.call(this);
173             $E.notifyObservers(this, "show", this.currentTab);
174 
175         },
176         /**
177          * 自动切换
178          * */
179         slide:function(){
180             var config = this.config,
181                 _this = this,
182                 intervalId,
183                 delayId;
184             J.array.forEach(this.tabs, function(tab, index, tabs){
185                 $E.on(tab.trigger, 'mouseover' , clear);
186                 $E.on(tab.sheet, 'mouseover' , clear);
187                 
188                 $E.on(tab.trigger, 'mouseout' , delay);
189                 $E.on(tab.sheet, 'mouseout' , delay);
190             });
191             start();
192             function start() {
193                 var i = _this.indexOf(_this.currentTab);
194                 if( i == -1 ) return;
195                 intervalId = window.setInterval(function(){
196                     var tab = _this.tabs[ ++i % _this.tabs.length ];
197                     if(tab){
198                         _this.select(tab);
199                     }
200                 },config['slideInterval']);
201             }
202             function clear() {
203                 window.clearTimeout(delayId);
204                 window.clearInterval(intervalId);   
205             }
206             function delay() {
207                 delayId = window.setTimeout(start,config['slideDelay']);
208             }
209         },
210         /**
211         * 切换到下一个
212         **/
213         next:function(){
214             var _this = this;
215             var i = _this.indexOf(_this.currentTab);
216             if(i == -1) return;
217             if(++i == _this.tabs.length){
218                 i = 0;
219             }
220             var tab = _this.tabs[i];
221             if(tab){
222                 _this.select(tab);
223             }
224         },
225         /**
226         * 切换到上一个
227         **/
228         prev:function(){
229             var _this = this;
230             var i = _this.indexOf(_this.currentTab);
231             if(i == -1) return;
232             if(--i == -1){
233                 i = _this.tabs.length - 1;
234             }
235             var tab = _this.tabs[i];
236             if(tab){
237                 _this.select(tab);
238             }
239         },
240         /**
241          * 获取tab在tabs数组中的索引
242          * @param {Object} tab  tab对象 {trigger:HTMLElement,sheet:HTMLElement}
243          * */
244         indexOf:function(tab){
245             for(var i = 0, len = this.tabs.length; i < len; i++){
246                 if(tab.trigger == this.tabs[i].trigger)
247                     return i;
248             }
249             return -1;
250         }
251     };
252 
253 });
