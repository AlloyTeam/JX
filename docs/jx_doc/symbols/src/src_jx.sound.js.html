102         init : function(url,autoLoadAndPlay,needEventSupport){
103             var context=this;
104             var autoLoadAndPlay = autoLoadAndPlay || false;
105             this._needEventSupport = !!needEventSupport;
106             this._url=url;
107             this._id=soundObjectList.length;
108             if(autoLoadAndPlay){
109                 var obj=new Audio();
110                 obj.id='audioSoundObject_'+this._id;
111                 this._onloadCallback=function(){
112                     context.play();
113                     context._obj.removeEventListener('canplay',context._onloadCallback,false);
114                     context._onloadCallback=null;
115                 }
116                 obj.addEventListener('canplay',this._onloadCallback,false);
117                 $D.id('sound_object_container').appendChild(obj);
118                 this._obj=obj;
119                 this.load.call(this,url);
120             }
121             soundObjectList.push(this);
122         },
123         load : function(url){
124             var context=this;
125             if(url) this._url=url;
126             else if(!this._url) return;
127             if(!this._obj){
128                 var obj=new Audio();
129                 obj.id='audioSoundObject_'+this._id;
130                 if(this._needEventSupport){
131                     obj.addEventListener('durationchange',function(){
132                         $E.notifyObservers(context,'durationchange');
133                     },false);
134                     obj.addEventListener('timeupdate',function(){
135                         $E.notifyObservers(context,'timeupdate');
136                     },false);
137                     obj.addEventListener('canplay',function(){
138                         $E.notifyObservers(context,'canplay');
139                     },false);
140                     obj.addEventListener('ended',function(){
141                         $E.notifyObservers(context,'ended');
142                     },false);
143                     obj.addEventListener('play',function(){
144                         $E.notifyObservers(context,'play');
145                     },false);
146                     obj.addEventListener('pause',function(){
147                         $E.notifyObservers(context,'pause');
148                     },false);
149                     obj.addEventListener('progress',function(){
150                         $E.notifyObservers(context,'progress');
151                     },false);
152                     obj.addEventListener('error',function(){
153                         $E.notifyObservers(context,'error');
154                     },false);
155                 }
156                 $D.id('sound_object_container').appendChild(obj);
157                 this._obj=obj;
158             }
159             this._obj.src=this._url;
160             if($B.mobileSafari) this._obj.load();
161         },
162         _setDirectVolume : function(value){
163             if(this._obj) this._obj.volume=value/100;
164         },
165         play : function(){
166             if(this._obj) this._obj.play();
167         },
168         pause : function(){
169             if(!this._obj) return;
170             if(this._onloadCallback){
171                 this._obj.removeEventListener('canplay',this._onloadCallback,false);
172                 this._onloadCallback=null;
173             }
174             this._obj.pause();
175         },
176         stop : function(){
177             if(this._obj){
178                 this._obj.pause();
179                 this.setPosition(0);
180             }
181         },
182         getDuration : function(){
183             if(!this._obj) return 0;
184             return this._obj.duration;
185         },
186         getPosition : function(){
187             if(!this._obj) return 0;
188             return this._obj.currentTime;
189         },
190         setPosition : function(value){
191             if(!this._obj) return false;
192             var obj=this._obj;
193             try{
194                 if(value>=0 && value<obj.duration){
195                     obj.currentTime=parseFloat(value);
196                     return true;
197                 }else{
198                     return false;
199                 }
200             }catch(e){
201                 return false;
202             }
203         },
204         buffered : function(){
205             if(!this._obj) return 0;
206             var obj=this._obj;
207             if(!obj.buffered.length) return 0;
208             return obj.buffered.end(0);
209         },
210         free : function(){
211             if(this._obj){
212                 var obj=this._obj;
213                 obj.pause();
214                 $D.id('sound_object_container').removeChild(obj);
215                 this._obj=null;
216             }
217             soundObjectList[this._id]=null;
218         }

311         /**
312          * @param {Bool} inject true|false 是否插入当前播放音频（播放后恢复）
313          */
314         play : function(inject, url){
315             if(this._obj) {
316                 
317                 if(inject) {
318                     this._obj._inject= true;
319                     var src, currentTime, _this= this;
320                     if(!this._obj.paused) {
321                         src= this._obj.src;
322                         if(!this._obj.ended) {
323                             currentTime= this._obj.currentTime+0.01;
324                         }
325                     }
326                     this.load(url);
327                     var innerCall= function() {
328                         _this._obj.removeEventListener("ended",innerCall,false);
329                         _this._obj._inject= false;
330                         if(src) {
331                             _this._obj.src= src;
332                             //ipad not support muted
333                             _this._obj.muted= true;
334                             _this._obj.load();
335                             if(currentTime) {
336                                 var setCall= function() {
337                                     try {
338                                         _this._obj.currentTime= currentTime;
339                                         _this._obj.muted= false;
340                                         _this._obj.play();
341                                     }catch(e) {
342                                         setTimeout(setCall, 1000);
343                                     }
344                                 }
345                                 setCall();
346                             }
347                         }
348                     };
349                     this._obj.addEventListener("ended",innerCall,false);
350                     setTimeout(function(){
351                         if(_this._obj._inject) {
352                             innerCall();
353                         }
354                     },3000);
355                     this._obj.play();
356                 }else {
357                     this._obj.play();
358                 }
359             }
360         },
361         pause : function(){
362             if(!this._obj) return;
363             this._obj.pause();
364         },
365         stop : function(){
366             if(this._obj){
367                 this._obj.pause();
368                 this.setPosition(0);
369             }
370         },

414         init : function(url,autoLoadAndPlay,needEventSupport){
415             this._fid=-1;
416             var context=this;
417             var autoLoadAndPlay = autoLoadAndPlay || false;
418             this._needEventSupport = !!needEventSupport;
419             this._url=url;
420             this._id=soundObjectList.length;
421             if(autoLoadAndPlay){
422                 var obj=$D.id('JxSwfSound_Flash');
423                 this._fid=obj.loadSound(this._url,-1,true); //url, callbackid, playWhileReady
424                 this._correctVolume();
425             }
426             soundObjectList.push(this);
427         },
428         load : function(url){
429             if(url) this._url=url;
430             else if(!this._url) return;
431             var obj=$D.id('JxSwfSound_Flash');
432             if(this._fid!=-1){
433                 obj.free(this._fid);
434             }
435             this._fid=obj.loadSound(this._url,this._needEventSupport?this._id:-1,false);
436             this._correctVolume();
437         },
438         _setDirectVolume : function(value){
439             if(this._fid==-1) return;
440             var obj=$D.id('JxSwfSound_Flash');
441             obj.setVolume(this._fid,value);
442         },

450             var obj=$D.id('JxSwfSound_Flash');

552     soundModeDetector = function(){
553         if(J.browser.chrome){
554             return 1;//chrome新版有问题, 播放不了audio
555         }else if(J.platform.iPad) {
556             return 4;
557         }else if(window.Audio && (new Audio).canPlayType('audio/mpeg')){
558             return 3; //以上浏览器的高版本支持audio对象播放mp3格式
559         }else if(J.browser.plugins.flash>=9){
560             return 1; //支持flash控件
561         }
562         /*else if(!!window.ActiveXObject && new ActiveXObject("WMPlayer.OCX.7")){
563             return 2; //支持wmp控件
564         }*/
565         else{
566             return 0; //一直很安静
567         }
568     };

592                 window.soundEventDispatcher=soundEventDispatcher;
593                 var d=$D.node('div',{
594                     id:'sound_object_container',

598                 embedSWF(option && option.path);
